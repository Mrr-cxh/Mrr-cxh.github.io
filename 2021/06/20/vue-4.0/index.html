  • 依赖注入 provide 和 inject

    • Parent2.vue

      <template>
  <div>
    <p>我是父组件2中的 data：{{ count }}</p>
    <Son2
      str="hello"
      class="active"
      title="xxx"
      id="fsfs"
      @click="
        () => {
          console.log('hello');
        }
      "
    />
  </div>
</template>

<script>
  import Son2 from "./Son2";
  export default {
    name: "Parent2",
    components: { Son2 },
    data() {
      return {
        count: 100,
      };
    },
    provide() {
      return {
        // 向下传递的时候，要写成函数带返回值的形式，这样后代才能动态获取
        // 只能写成箭头函数，不能写成普通函数，因为普通函数的this指向的是 return{}
        count: () => this.count,
        changeCount: this.changeCount,
      };
    },
    methods: {
      changeCount(newCount) {
        this.count = newCount;
      },
    },
  };
</script>

    • Son2.vue

      <template>
  <div>
    <p>子组件2使用父组件2中的 data：{{ count() }}</p>
    <p>父组件传递的str props：{{ $attrs.str }}</p>
    <GrandSon2 />
  </div>
</template>

<script>
  import GrandSon2 from "./GrandSon2";
  export default {
    name: "Son2",
    components: { GrandSon2 },
    inject: ["count"],
    created() {
      // 获取当前组件的所有 html 属性,就是父组件使用我的时候标签内传递的所有属性,class除外
      console.log(this.$attrs);
      // 获取传递的事件
      console.log(this.$listeners);
    },
  };
</script>

    • GrandSon2.vue

      <template>
  <div>
    <p>子子组件2使用父父组件2中的 data：{{ count() }}</p>
    <button @click="handleClick">使用inject获取祖先的修改事件并执行</button>
  </div>
</template>

<script>
  export default {
    name: "Parent2",
    inject: ["count", "changeCount"],
    methods: {
      handleClick() {
        this.changeCount(5000);
        // 如果provide只是传值的话只是修改了 Parent2,剩下的两个组件没有被修改
        // 把 provide 的 count 改成了函数，就会全部跟着修改
      },
    },
  };
</script>

# 函数式组件

函数组件和普通组件没区别
如果一个组件只接收 props 没有自己的 data，没有 data 的组件可以称为无状态组件，此时可以将这个组件设置成函数组件
两种组件使用和创建都一样，只不过函数组件渲染会快

  • 单文件形式的组件

    • RenderDemo.vue

      <template>
  <div>
    <h3>函数组件</h3>
    <FunComponent :list="list" :handleClick="handleClick">
      <h4>测试 slot</h4>
    </FunComponent>
  </div>
</template>

<script>
  import FunComponent from "./FunComponent";
  export default {
    name: "RenderDemo",
    components: {
      FunComponent,
    },
    data() {
      return {
        list: [1, 2, 3],
      };
    },
    methods: {

        console.log("传递事件");
      },
    },
  };

  • plugins 文件夹下 index.js 文件

    import Vue from "vue";
// 制作一个插件
const myPlugin = {
  install() {
    // 组件内全局的变量
    Vue.prototype.$num = 1000;

    // 组件的全局指令 v-focus
    Vue.directive("focus", {
      inserted: function (el, binding) {
        // binding是一个对象，binding.value代表所绑定的值
        binding.value && el.focus();
      },
    });
  },
};
// use方法主要就是执行了 参数内的 install 方法
// install 就相当于是为 use 专门而生的
Vue.use(myPlugin);
// export default myPlugin;

    // 制作一个处理扁平化数组的函数,将其转换成树
const getTree = (authMenus) => {
  // console.log(authMenus);
  const menu = [];
  const sourceMap = {};
  authMenus.forEach((m) => {
    m.children = [];
    sourceMap[m.id] = m;
    if (m.pid === -1) {
      // 根目录
      menu.push(m);
    } else {
      /* const parentMenu = authMenus.find((item) => item.id === m.pid);
        parentMenu && parentMenu.children.push(m); */
      sourceMap[m.pid].children.push(m);
    }
  });
  return menu;
};
export { getTree };

        // 直接根据地址栏的 hash 修改 currentRoute
        this.currentRoute = this.routeMap[hash.slice(1)];
      }
    });
  }

      const data = {
        count: 100,
        message: "hello world",
        obj: { a: 10, b: 20 },
        list: [1, 2, 3, 4],
      };

      // 给数组制作一个原型对象
      const arrayMethods = ["push", "pop", "shift", "unshift", "splice"];
      // ? * ? 数组的原型无法拷贝,所以我们的新原型内没有其他的方法
      const newArrayPrototype = {};
      // 让数组push的时候，用这个新创建的代替原来的执行
      arrayMethods.forEach((methodName) => {
        Object.defineProperty(newArrayPrototype, methodName, {
          enumerable: false,
          value: function (...arg) {
            Array.prototype[methodName].call(this, ...arg);
            // 当 push 或其他可更新方法执行的时候 传递的参数也需要监听
            // 应该循环 arg 对所有的参数进行监听
            // 用了笨办法 直接重新监听整个对象
            observer(this);
            console.log("更新视图");
          },
        });
      });

      // 将 data 对象制作成一个响应式对象
      // 就是当你修改 data 的任何内容的时候都可以监听到，根据变化更新视图
      // 需要制作一个方法，observer
      observer(data);

      function isObject(target) {
        return !(typeof target !== "object" || target === null);
      }

      function observer(target) {
        // console.log(target);
        // 当 target 是非对象数据的时候 直接返回,
        // 是对象数据那就需要对对象下的所有属性进行监听
        // instanceof 用来检测一个数据是由哪个构造函数创建的
        if (!isObject(target)) {
          return target;
        }

        // 是数组的话
        if (Array.isArray(target)) {
          // 构造函数才有 prototype
          // 构造函数创建出来的用 __proto__ 去修改
          target.__proto__ = newArrayPrototype;
          // 只能修改 该数组的原型对象 并不是所有数组的
        }

        // 遍历对象下的所有属性
        for (const key in target) {
          // 对所有属性进行监听 就是使用 Object.defineProperty 对所有属性进行属性描述
          // 将这个功能定义成另外一个方法
          defineReactive(target, key);
          // 判断是对象 还需要继续 observer
          // observer(key);
          if (isObject(target[key])) {
            observer(target[key]);
          }
        }
      }

      function defineReactive(target, key) {
        let value = target[key];
        Object.defineProperty(target, key, {
          get() {
            return value;
          },
          set(newValue) {
            if (newValue !== value) {
              value = newValue;
              // 当 newValue 是一个对象数据的时候，我们对这个数据并没有监听
              if (isObject(newValue)) {
                observer(newValue);
              }
              console.log("更新视图");
            }
          },
        });
      }
